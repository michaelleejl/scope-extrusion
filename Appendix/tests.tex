\begin{metaocaml}
let () =  Trx.set_with_stack_mark
  {Trx.stackmark_region_fn =
      fun body ->
        let module M = struct type _ Effect.t += E: unit t end in
        try body (fun () -> try perform M.E; true with Unhandled _ -> false)
        with effect M.E, k -> continue k ()}
\end{metaocaml}
